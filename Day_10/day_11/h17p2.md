# Binary Tree Right Side View
https://leetcode.com/problems/binary-tree-right-side-view/description/
```
python
class Solution:
    def rightSideView(self, root):
        if not root:
            return []
        ans = []
        q = deque([root])
        while q:
            size = len(q)
            for i in range(size):
                node = q.popleft()
                if node.left:
                    q.append(node.left)
                if node.right:
                    q.append(node.right)
                # Last node of the level
                if i == size - 1:
                    ans.append(node.val)
        return ans
```
